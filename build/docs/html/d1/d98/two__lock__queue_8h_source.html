109
110 while (true) {

114 // Check if tail hasn't been changed by another thread
115 if (last == tail_.load(std::memory_order_acquire)) {
116 if (next == nullptr) {
117 // Last node is the actual tail
118 if (last->next.compare_exchange_weak(next, new_node,

155 bool dequeue(T& item) {
156 while (true) {
157 Node* first = head_.load(std::memory_order_acquire);
158 Node* last = tail_.load(std::memory_order_acquire);
159 Node* next = first->next.load(std::memory_order_acquire);
160
161 // Check if head hasn't been changed by another thread
162 if (first == head_.load(std::memory_order_acquire)) {
163 if (first == last) {
164 if (next == nullptr) {
165 // Queue is empty
166 return false;
167 }
168 // Tail is lagging, help advance it
169 tail_.compare_exchange_weak(last, next,
170 std::memory_order_release,
171 std::memory_order_relaxed);
172 } else {
173 // Queue has elements
