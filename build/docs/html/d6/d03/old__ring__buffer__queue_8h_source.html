114 bool enqueue(T item) {
115 size_t write = write_pos_.load(std::memory_order_relaxed);
116
117 while (true) {
118 size_t read = read_pos_.load(std::memory_order_acquire);
119
120 // Check if queue is full
121 if (write - read >= capacity_) {
122 return false;
123 }
124
125 // Try to claim the write position
126 if (write_pos_.compare_exchange_weak(write, write + 1,
127 std::memory_order_acq_rel)) {
128

134 }
