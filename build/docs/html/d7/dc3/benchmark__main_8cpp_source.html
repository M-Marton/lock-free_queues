81Config parse_args(int argc, char* argv[]) {
82 Config cfg;
83
84 static struct option long_options[] = {
85 {"type", required_argument, 0, 't'},
86 {"producers", required_argument, 0, 'p'},
87 {"consumers", required_argument, 0, 'c'},
88 {"items", required_argument, 0, 'i'},
89 {"size", required_argument, 0, 's'},
90 {"help", no_argument, 0, 'h'},
91 {0, 0, 0, 0}
92 };
93
94 int opt;
95 while ((opt = getopt_long(argc, argv, "t:p:c:i:s:h", long_options, nullptr)) != -1) {
96 switch (opt) {
97 case 't':
98 cfg.queue_type = optarg;
99 break;
100 case 'p':
101 cfg.producers = std::stoul(optarg);
102 if (cfg.producers == 0) {
103 std::cerr << "Error: Producers must be > 0\n";
104 exit(1);
105 }
106 break;
107 case 'c':
108 cfg.consumers = std::stoul(optarg);
109 if (cfg.consumers == 0) {
110 std::cerr << "Error: Consumers must be > 0\n";
111 exit(1);
112 }
113 break;
114 case 'i':
115 cfg.items = std::stoul(optarg);
116 if (cfg.items == 0) {
117 std::cerr << "Error: Items must be > 0\n";
118 exit(1);
119 }
120 break;
121 case 's':
122 cfg.capacity = std::stoul(optarg);
123 if (cfg.capacity == 0) {
124 std::cerr << "Error: Capacity must be > 0\n";
125 exit(1);
126 }
127 break;
128 case 'h':
129 cfg.help = true;
130 break;
131 default:
132 cfg.help = true;
133 break;
134 }
135 }
