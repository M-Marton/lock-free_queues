94 void add(uint64_t latency_ns) {
95 if (count == 0 || latency_ns < min) min = latency_ns;
96 if (count == 0 || latency_ns > max) max = latency_ns;
97 sum += latency_ns;
98 sorted_latencies.push_back(latency_ns);
99 count++;
100 }
101
106 double avg() const {
107 return count > 0 ? static_cast<double>(sum) / count : 0.0;
108 }
109
115 uint64_t percentile(double percentile) const {
116 if (sorted_latencies.empty()) return 0;
117 size_t idx = static_cast<size_t>(percentile / 100.0 * sorted_latencies.size());
118 idx = std::min(idx, sorted_latencies.size() - 1);
119 return sorted_latencies[idx];
120 }
121
125 void sort() {
126 std::sort(sorted_latencies.begin(), sorted_latencies.end());
127 }
128
132 void reset() {
133 count = 0;
134 min = 0;
135 max = 0;
136 sum = 0;
137 sorted_latencies.clear();
138 }

261 void warmup() {
262 constexpr size_t WARMUP_ITEMS = 10000;
263 std::atomic<bool> warmup_stop{false};
264
265 // Simple warm-up: single producer, single consumer
266 std::thread producer([this, &warmup_stop]() {
267 size_t produced = 0;
268 while (produced < WARMUP_ITEMS && !warmup_stop) {
269 if (queue_.enqueue(produced)) {
270 produced++;
271 }
272 }
273 });
274
275 std::thread consumer([this, &warmup_stop]() {
276 uint64_t item;
277 size_t consumed = 0;
278 while (consumed < WARMUP_ITEMS && !warmup_stop) {
279 if (queue_.dequeue(item)) {
280 consumed++;
281 }
282 }
283 });
284
285 producer.join();
286 consumer.join();
287
288 // Clear the queue
289 uint64_t item;
290 while (queue_.dequeue(item)) {}
291
292 std::cout << "Warm-up complete." << std::endl;
293 }

305 void producer_worker(size_t thread_id) {
306 uint64_t produced = 0;
307
308 while (produced < items_per_producer_) {
309 auto before = std::chrono::high_resolution_clock::now();
310
311 if (queue_.enqueue(produced)) {

313 auto latency_ns = std::chrono::duration_cast<std::chrono::nanoseconds>(

316 // LTTng tracepoint

318 tracepoint(benchmark_mpmc, queue_operation,

321
322 // Record statistics

324 enqueue_stats_.add(latency_ns);
325

327 total_enqueued_++;

345 void consumer_worker(size_t thread_id) {
